\begin{algorithm}
  \caption{Deterministic \dssta\ considering the static power}
  \alab{temperature-solution}
  \begin{algorithmic}[1]
    \vspace{0.4em}

    \Require{$\mP_\dynamic \in \real^{\nprocs \times \nsteps}$ and $\vu \in \real^\nprocs$}
    \Ensure{$\mQ \in \real^{\nprocs \times \nsteps}$ and $\mP \in \real^{\nprocs \times \nsteps}$}

    \vspace{0.5em}

    \Let{$\mQ$}{$\mQ_\ambient$}

    \Repeat
      \Let{$\mP$}{$\mP_\dynamic + \mP_\static(\vu, \mQ)$}
      \Let{$\mQ$}{call {\bf \aref{temperature-solution-without-leakage}} for $\mP$}
    \Until{a stopping condition is satisfied}

    \vspace{0.4em}
  \end{algorithmic}
\end{algorithm}
