  \caption{Deterministic analysis with leakage. \alabel{deterministic-solution-with-leakage}}
  \begin{algorithmic}[1]
    \vspace{0.4em}

    \Require{A vector of parameters $\vU$.}
    \Ensure{The corresponding temperature profile $\mQ$.}

    \Let{$\mQ$}{$\mQ_\ambient$}

      \Let{$\mQ$}{run {\bf Algorithm 1} for $\mP$}
    \Until{a stopping condition is met}

    \vspace{0.4em}
  \end{algorithmic}
